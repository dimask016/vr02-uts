string GetCurrentObjectiveHint()
{
if (!hasIDCard) return "🔵 Go to BLUE station …";
if (!isAlarmActivated) return "🔴 Go to RED station …";
if (!isDoorOpened) return "🟢 Go to GREEN station …";
if (!isEvacuationComplete) return "🟡 Go to YELLOW station …";
return "🏆 Evacuation Complete!";
}		 GetCurrentObjectiveHint() Mengembalikan petunjuk teks berdasarkan status progres. Digunakan untuk mengisi status bar setelah aksi atau saat inisialisasi. Membantu pemain tahu ke mana harus pergi.
