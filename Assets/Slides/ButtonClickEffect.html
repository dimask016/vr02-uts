🎨 UI Effect · Runtime · 54 baris

Fungsi utama: Memberikan efek visual interaktif pada tombol UI (berbasis Image) saat kursor di-hover, ditekan, atau dilepas. Mengubah skala dan warna secara halus menggunakan Vector3.Lerp dan Color.Lerp dengan durasi yang dapat diatur.

public class ButtonClickEffect : MonoBehaviour,
IPointerEnterHandler, IPointerExitHandler,
IPointerDownHandler, IPointerUpHandler		 Deklarasi Kelas Mewarisi MonoBehaviour dan mengimplementasikan empat antarmuka IPointer… untuk menangkap event pointer dari sistem UI Unity.
public float hoverScale = 1.1f;
public float pressScale = 0.95f;
public float duration = 0.1f;
public Color hoverColor = new Color(0.4f, 0.8f, 1f);
public Color pressColor = new Color(0.1f, 0.4f, 0.7f);		 Parameter Efek Nilai yang dapat diatur di Inspector: hoverScale (skala saat kursor di atas), pressScale (skala saat ditekan), duration (kecepatan transisi), dan warna untuk status hover/press.
private Image image;
private Vector3 originalScale;
private Color originalColor;
private bool isHovering = false;
private bool isPressed = false;		 Variabel Internal Menyimpan referensi komponen Image, skala dan warna awal tombol, serta status boolean untuk mengetahui apakah sedang di-hover atau ditekan.
void Start()
{
image = GetComponent<Image>();
originalScale = transform.localScale;
if (image != null) originalColor = image.color;
}		 Start() Inisialisasi: mengambil komponen Image dari GameObject, menyimpan skala asli dan warna asli (jika ada) untuk digunakan sebagai basis perhitungan efek.
void Update()
{
Vector3 targetScale = originalScale;
if (isPressed) targetScale = originalScale * pressScale;
else if (isHovering) targetScale = originalScale * hoverScale;
transform.localScale = Vector3.Lerp(transform.localScale, targetScale, Time.deltaTime / duration);
…
}		 Update() Setiap frame, tentukan skala target berdasarkan status (press > hover > normal). Kemudian lerp skala dan warna menuju target dengan kecepatan yang ditentukan duration.
if (image != null)
{
Color targetColor = originalColor;
if (isPressed) targetColor = pressColor;
else if (isHovering) targetColor = hoverColor;
image.color = Color.Lerp(image.color, targetColor, Time.deltaTime / duration);
}		 Update() – Warna Bagian dari Update yang menangani perubahan warna tombol. Sama seperti skala, warna juga di-lerp menuju target (pressColor, hoverColor, atau originalColor).
public void OnPointerEnter(PointerEventData eventData)
{
isHovering = true;
}		 OnPointerEnter() Dipanggil saat pointer (kursor) memasuki area tombol. Mengatur isHovering = true sehingga Update akan menerapkan efek hover.
public void OnPointerExit(PointerEventData eventData)
{
isHovering = false;
}		 OnPointerExit() Dipanggil saat pointer keluar dari area tombol. Mengatur isHovering = false sehingga tombol kembali ke skala dan warna normal.
public void OnPointerDown(PointerEventData eventData)
{
isPressed = true;
}		 OnPointerDown() Dipanggil saat tombol ditekan (mouse atau sentuhan). Mengatur isPressed = true sehingga efek press (skala mengecil, warna gelap) aktif.
public void OnPointerUp(PointerEventData eventData)
{
isPressed = false;
}		 OnPointerUp() Dipanggil saat tombol dilepas. Mengatur isPressed = false sehingga efek press hilang dan tombol kembali ke status hover atau normal.
// (seluruh kode) Cara Kerja Efek ini bekerja secara independen dari logika tombol (misalnya Button.onClick). Hanya menangani aspek visual. Dapat dipasang pada objek yang memiliki komponen Image (biasanya tombol UI).
